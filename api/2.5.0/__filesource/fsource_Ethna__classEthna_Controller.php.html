  848.     /**
  849.      *  フレームワークの処理を開始する
  850.      *
  851.      *  @access public
  852.      *  @param  mixed   $default_action_name    指定のアクション名
  853.      *  @param  mixed   $fallback_action_name   アクション名が決定できなかった場合に実行されるアクション名
  854.      *  @param  bool    $enable_filter  フィルタチェインを有効にするかどうか
  855.      *  @return mixed   0:正常終了 Ethna_Error:エラー
  856.      */
  857.     function trigger($default_action_name = "", $fallback_action_name = "", $enable_filter = true)
  858.     {
  859.         // フィルターの生成
  860.         if ($enable_filter) {
  861.             $this->_createFilterChain();
  862.         }
  863.  
  864.         // 実行前フィルタ
  865.         for ($i = 0; $i < count($this->filter_chain); $i++) {
  866.             $r = $this->filter_chain[$i]->preFilter();
  867.             if (Ethna::isError($r)) {
  868.                 return $r;
  869.             }
  870.         }
  871.  
  872.         // trigger
  873.         switch ($this->getGateway()) {
  874.         case GATEWAY_WWW:
  875.             $this->_trigger_WWW($default_action_name, $fallback_action_name);
  876.             break;
  877.         case GATEWAY_CLI:
  878.             $this->_trigger_CLI($default_action_name);
  879.             break;
  880.         case GATEWAY_XMLRPC:
  881.             $this->_trigger_XMLRPC();
  882.             break;
  883.         case GATEWAY_SOAP:
  884.             $this->_trigger_SOAP();
  885.             break;
  886.         }
  887.  
  888.         // 実行後フィルタ
  889.         for ($i = count($this->filter_chain) - 1; $i >= 0; $i--) {
  890.             $r = $this->filter_chain[$i]->postFilter();
  891.             if (Ethna::isError($r)) {
  892.                 return $r;
  893.             }
  894.         }
  895.     }
  896.  
  897.     /**
  898.      *  フレームワークの処理を実行する(WWW)
  899.      *
  900.      *  引数$default_action_nameに配列が指定された場合、その配列で指定された
  901.      *  アクション以外は受け付けない(指定されていないアクションが指定された
  902.      *  場合、配列の先頭で指定されたアクションが実行される)
  903.      *
  904.      *  @access private
  905.      *  @param  mixed   $default_action_name    指定のアクション名
  906.      *  @param  mixed   $fallback_action_name   アクション名が決定できなかった場合に実行されるアクション名
  907.      *  @return mixed   0:正常終了 Ethna_Error:エラー
  908.      */
  909.     function _trigger_WWW($default_action_name = "", $fallback_action_name = "")
  910.     {
  911.         // アクション名の取得
  912.         $action_name = $this->_getActionName($default_action_name, $fallback_action_name);
  913.  
  914.         // マネージャ実行チェック
  915.         $this->_ethnaManagerEnabledCheck($action_name);
  916.  
  917.         // アクション定義の取得
  918.         $action_obj =& $this->_getAction($action_name);
  919.         if (is_null($action_obj)) {
  920.             if ($fallback_action_name != "") {
