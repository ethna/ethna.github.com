  189.                 if ($this->_isSubclassOf($class_name, 'Ethna_ViewClass')) {

  191.                     $class_list[$view_name] = array(
  192.                         'template_file' => $this->ctl->_getForwardPath($view_name),
  193.                         'view_class' => $class_name,
  194.                         'view_class_file' => $this->ctl->getDefaultViewPath($view_name),
  195.                     );

  199.  
  200.         if (count($class_list) == 0) {

  224.  
  225.         $dh = opendir($view_dir);

  229.                 if ($file != '.' && $file != '..' && is_dir($path)) {

  238.                 include_once($path);
  239.                 $class_list = $this->__analyzeViewScript($path);

  248.     }
