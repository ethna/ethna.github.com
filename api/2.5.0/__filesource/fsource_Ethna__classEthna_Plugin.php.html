  133.     function &_getPlugin($type, $name)
  134.     {
  135.         if (isset($this->obj_registry[$type]) == false) {
  136.             $this->obj_registry[$type] = array();
  137.  
  138.             // プラグインの親クラスを(存在すれば)読み込み
  139.             foreach ($this->appid_list as $appid) {
  140.                 list($class, $dir, $file) = $this->getPluginNaming($type, null, $appid);
  141.                 $this->_includePluginSrc($class, $dir, $file, true);
  142.             }
  143.         }
  144.  
  145.         // key がないときはプラグインをロードする
  146.         if (array_key_exists($name, $this->obj_registry[$type]) == false) {
  147.             $this->_loadPlugin($type, $name);
  148.         }
  149.  
  150.         // null のときはロードに失敗している
  151.         if (is_null($this->obj_registry[$type][$name])) {
  152.             return Ethna::raiseWarning('plugin [type=%s, name=%s] is not found',
  153.                 E_PLUGIN_NOTFOUND, $type, $name);
  154.         }
  155.  
  156.         // プラグインのインスタンスを返す
  157.         return $this->obj_registry[$type][$name];
  158.     }
  159.  
  160.     /**
  161.      *  プラグインをincludeしてnewし，レジストリに登録
  162.      *
  163.      *  @access private
  164.      *  @param  string  $type   プラグインの種類
  165.      *  @param  string  $name   プラグインの名前
  166.      */
  167.     function _loadPlugin($type, $name)
  168.     {
  169.         // プラグインのファイル名を取得
  170.         $plugin_src = $this->_getPluginSrc($type, $name);
  171.         if (is_null($plugin_src)) {
  172.             $this->obj_registry[$type][$name] = null;
  173.             return;
  174.         }
  175.         list($plugin_class, $plugin_dir, $plugin_file) = $plugin_src;
  176.  
  177.         // プラグインのファイルを読み込み
  178.         $r =& $this->_includePluginSrc($plugin_class, $plugin_dir, $plugin_file);
  179.         if (Ethna::isError($r)) {
  180.             $this->obj_registry[$type][$name] = null;
  181.             return;
  182.         }
  183.  
  184.         // プラグイン作成
  185.         $instance =& new $plugin_class($this->controller, $type, $name);
  186.         if (is_object($instance) == false
  187.             || strcasecmp(get_class($instance), $plugin_class) != 0) {
  188.             $this->logger->log(LOG_WARNING, 'plugin [%s::%s] instantiation failed', $type, $name);
  189.             $this->obj_registry[$type][$name] = null;
  190.             return;
  191.         }
  192.         $this->obj_registry[$type][$name] =& $instance;
  193.     }
  194.

  234.     /**
  235.      *  プラグインのクラス名、ディレクトリ、ファイル名を決定
  236.      *
  237.      *  @access public
  238.      *  @param  string  $type   プラグインの種類
  239.      *  @param  string  $name   プラグインの名前 (nullのときは親クラス)
  240.      *  @param  string  $appid  アプリケーションID
  241.      *  @return array   プラグインのクラス名、ディレクトリ、ファイル名の配列
  242.      *  @todo   class factoryのnaming ruleと整合させる

  244.     function getPluginNaming($type, $name, $appid)
  245.     {
  246.         if ($appid == 'Ethna') {
  247.             if ($name === null) {
  248.                 $ext = 'php';
  249.                 $dir = ETHNA_BASE . "/class/Plugin";
  250.                 $class = "Ethna_Plugin_{$type}";
  251.             } else {
  252.                 $ext = 'php';
  253.                 $dir = ETHNA_BASE . "/class/Plugin/{$type}";
  254.                 $class = "Ethna_Plugin_{$type}_{$name}";
  255.             }
  256.         } else {
  257.             if ($name === null) {
  258.                 $ext = $this->controller->getExt('php');
  259.                 $dir = $this->controller->getDirectory('plugin');
  260.                 $class = "{$appid}_Plugin_{$type}";
  261.             } else {
  262.                 $ext = $this->controller->getExt('php');
  263.                 $dir = $this->controller->getDirectory('plugin') . "/{$type}";
  264.                 $class = "{$appid}_Plugin_{$type}_{$name}";
  265.             }
  266.         }
  267.  
  268.         $file  = "{$class}.{$ext}";
  269.         return array($class, $dir, $file);
  270.     }

  272.     /**
  273.      *  プラグインのソースを include する
  274.      *
  275.      *  @access private
  276.      *  @param  string  $class  クラス名
  277.      *  @param  string  $dir    ディレクトリ名
  278.      *  @param  string  $file   ファイル名
  279.      *  @param  bool    $parent 親クラスかどうかのフラグ
  280.      *  @return true|Ethna_Error
  281.      */
  282.     function &_includePluginSrc($class, $dir, $file, $parent = false)
  283.     {
  284.         $true = true;
  285.         if (class_exists($class)) {
  286.             return $true;
  287.         }
  288.  
  289.         $file = $dir . '/' . $file;
  290.         if (file_exists($file) === false) {
  291.             if ($parent === false) {
  292.                 return Ethna::raiseWarning('plugin file is not found: [%s]',
  293.                                            E_PLUGIN_NOTFOUND, $file);
  294.             } else {
  295.                 return $true;
  296.             }
  297.         }
  298.  
  299.         include_once $file;
  300.  
  301.         if (class_exists($class) === false) {
  302.             if ($parent === false) {
  303.                 return Ethna::raiseWarning('plugin class [%s] is not defined',
  304.                     E_PLUGIN_NOTFOUND, $class);
  305.             } else {
  306.                 return $true;
  307.             }
  308.         }
  309.  
  310.         if ($parent === false) {
  311.             $this->logger->log(LOG_DEBUG, 'plugin class [%s] is defined', $class);
  312.         }
  313.         return $true;
  314.     }

  316.     /**
  317.      *  アプリケーション, Ethna の順でプラグインのソースを検索する
  318.      *
  319.      *  @access private
  320.      *  @param  string  $type   プラグインの種類
  321.      *  @param  string  $name   プラグインの名前
  322.      */
  323.     function _searchPluginSrc($type, $name)
  324.     {
  325.         // コントローラで指定されたアプリケーションIDの順に検索
  326.         foreach ($this->appid_list as $appid) {
  327.             list($class, $dir, $file) = $this->getPluginNaming($type, $name, $appid);
  328.             if (class_exists($class)) {
  329.                 // すでにクラスが存在する場合は特別にスキップ
  330.                 if (isset($this->src_registry[$type]) == false) {
  331.                     $this->src_registry[$type] = array();
  332.                 }
  333.                 $this->src_registry[$type][$name] = array($class, null, null);
  334.                 return;
  335.             }
  336.             if (file_exists("{$dir}/{$file}")) {
  337.                 $this->logger->log(LOG_DEBUG, 'plugin file is found in search: [%s]',
  338.                                    "{$dir}/{$file}");
  339.                 if (isset($this->src_registry[$type]) == false) {
  340.                     $this->src_registry[$type] = array();
  341.                 }
  342.                 $this->src_registry[$type][$name] = array($class, $dir, $file);
  343.                 return;
  344.             }
  345.         }
  346.  
  347.         // 見つからなかった場合 (nullで記憶しておく)
  348.         $this->logger->log(LOG_WARNING, 'plugin file for [type=%s, name=%s] is not found in search', $type, $name);
  349.         $this->src_registry[$type][$name] = null;
  350.     }
  351.  
  352.     /**
  353.      *  プラグインの種類 ($type) をすべて検索する
  354.      *
  355.      *  @access public
  356.      *  @return array 
  357.      */
  358.     function searchAllPluginType()
  359.     {
  360.         $type_list = array();
  361.         foreach (array_reverse($this->appid_list) as $appid) {
  362.             list(, $dir, ) = $this->getPluginNaming('', null, $appid);
  363.             if (is_dir($dir) == false) {
  364.                 continue;
  365.             }
  366.             $dh = opendir($dir);
  367.             if (is_resource($dh) == false) {
  368.                 continue;
  369.             }
  370.             while (($type_dir = readdir($dh)) !== false) {
  371.                 if ($type_dir{0} != '.' && is_dir("{$dir}/{$type_dir}")) {
  372.                     $type_list[$type_dir] = 0;
  373.                 }
  374.             }
  375.             closedir($dh);
  376.         }
  377.         return array_keys($type_list);
  378.     }
  379.  
  380.     /**
  381.      *  指定された $type のプラグイン (のソース) をすべて検索する
  382.      *
  383.      *  @access public
  384.      *  @param  string  $type   プラグインの種類
  385.      */
  386.     function searchAllPluginSrc($type)
  387.     {
  388.         // 後で見付かったもので上書きするので $this->appid_list の逆順とする
  389.         $name_list = array();
  390.         foreach (array_reverse($this->appid_list) as $appid) {
  391.             //  クラス名として許可された文字であればOKとする
  392.             //  アンダーバーを拒む理由はないし、命名規約からも禁止されていない
  393.             //  @see http://ethna.jp/ethna-document-dev_guide-plugin.html
  394.             //  @see http://www.php.net/manual/en/language.variables.php
  395.             list($class_regexp, $dir, $file_regexp) = $this->getPluginNaming($type, '([a-zA-Z0-9_\x7f-\xff]+)', $appid);
  396.  
  397.             //ディレクトリの存在のチェック
  398.             if (is_dir($dir) == false) {
  399.                 // アプリ側で見付からないのは正常
  400.                 continue;
  401.             }
  402.  
  403.             // ディレクトリを開く
  404.             $dh = opendir($dir);
  405.             if (is_resource($dh) == false) {
  406.                 $this->logger->log(LOG_DEBUG, 'cannot open plugin directory: [%s]', $dir);
  407.                 continue;
  408.             }
  409.             $this->logger->log(LOG_DEBUG, 'plugin directory opened: [%s]', $dir);
  410.  
  411.             // 条件にあう $name をリストに追加
  412.             while (($file = readdir($dh)) !== false) {
  413.                 if (preg_match('#^'.$file_regexp.'$#', $file, $matches)
  414.                     && file_exists("{$dir}/{$file}")) {
  415.                     $name_list[$matches[1]] = true;
  416.                 }
  417.             }
  418.  
  419.             closedir($dh);
  420.         }
  421.  
  422.         foreach (array_keys($name_list) as $name) {
  423.             // 冗長だがもう一度探しなおす
  424.             $this->_searchPluginSrc($type, $name);
  425.         }
  426.     }
