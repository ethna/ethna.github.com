  148.     /**
  149.      * @access private
  150.      */
  151.     function &_makePackage($skelfile, $setting, $workdir)
  152.     {
  153.         if (Ethna_Util::mkdir($workdir, 0755) === false) {
  154.             return Ethna::raiseError("failed making working dir: $workdir.");
  155.         }
  156.  
  157.         // プラグインの元ファイルを作成
  158.         $rfp = fopen($skelfile, "r");
  159.         if ($rfp == false) {
  160.             return Ethna::raiseError("failed open skelton file: $skelfile.");
  161.         }
  162.         $outputfile = $setting['filename'];
  163.         $wfp = fopen("$workdir/$outputfile", "w");
  164.         if ($rfp == false) {
  165.             fclose($rfp);
  166.             return Ethna::raiseError("failed creating working file: $outputfile.");
  167.         }
  168.         for (;;) {
  169.             $s = fread($rfp, 4096);
  170.             if (strlen($s) == 0) {
  171.                 break;
  172.             }
  173.             foreach ($setting['macro'] as $k => $v) {
  174.                 $s = preg_replace("/{\\\$$k}/", $v, $s);
  175.             }
  176.             fwrite($wfp, $s);
  177.         }
  178.         fclose($wfp);
  179.         fclose($rfp);
  180.  
  181.         // package.xml を作る
  182.         $pkgconfig = array(
  183.             'packagedirectory' => $workdir,
  184.             'outputdirectory' => $workdir,
  185.             'ignore' => array('CVS/', '.cvsignore', '.svn/',
  186.                               'package.xml', 'package.ini', $setting['pkgname'].'-*.tgz'),
  187.             'filelistgenerator' => 'file',
  188.             'changelogoldtonew' => false,
  189.             );
  190.  
  191.         $packagexml =& new PEAR_PackageFileManager2();
  192.  
  193.         $pkgconfig = array_merge($pkgconfig, $setting['config']);
  194.         $packagexml->setOptions($pkgconfig);
  195.  
  196.         $packagexml->setPackage($setting['pkgname']);
  197.         $packagexml->setSummary($setting['summary']);
  198.         $packagexml->setNotes($setting['notes']);
  199.         $packagexml->setDescription($setting['description']);
  200.         $packagexml->setChannel($setting['channel']);
  201.         $packagexml->setAPIVersion($setting['version']);
  202.         $packagexml->setReleaseVersion($setting['version']);
  203.         $packagexml->setReleaseStability($setting['state']);
  204.         $packagexml->setAPIStability($setting['state']);
  205.         $packagexml->setPackageType('php');
  206.         foreach ($setting['maintainers'] as $m) {
  207.             $packagexml->addMaintainer($m['role'], $m['user'], $m['name'],
  208.                                        $m['email'], $m['active']);
  209.         }
  210.         $packagexml->setLicense($setting['license']['name'],
  211.                                 $setting['license']['uri']);
  212.  
  213.         $packagexml->addRole('css', 'php');
  214.         $packagexml->addRole('tpl', 'php');
  215.         $packagexml->addRole('ethna', 'php');
  216.         $packagexml->addRole('sh', 'script');
  217.         $packagexml->addRole('bat', 'script');
  218.  
  219.         $packagexml->setPhpDep('4.1.0');
  220.         $packagexml->setPearinstallerDep('1.3.5');
  221.  
  222.         $packagexml->generateContents();
  223.  
  224.         foreach ($setting['callback'] as $method => $params) {
  225.             $r = call_user_func_array(array(&$packagexml, $method), $params);
  226.         }
  227.  
  228.         $r = $packagexml->writePackageFile();
  229.         if (PEAR::isError($r)) {
  230.             return Ethna::raiseError($r->getMessage, $r->getCode());
  231.         }
  232.  
  233.         // package を作る
  234.         $r = $this->pear->_run('package', array(), array("$workdir/package.xml"));
  235.         if (PEAR::isError($r)) {
  236.             return Ethna::raiseError($r->getMessage, $r->getCode());
  237.         }
  238.  
  239.         if (Ethna_Util::purgeDir($workdir) === false) {
  240.             return Ethna::raiseError("failed cleaning up working dir: $workdir.");
  241.         }
  242.     }
  243.     // }}}

  246.     /**
  247.      * @access private

  249.     function &_parseArgList()
  250.     {
  251.         $r =& $this->_getopt(array('inifile=', 'skelfile=', 'workdir='));
  252.         if (Ethna::isError($r)) {
  253.             return $r;
  254.         }
  255.         list($opt_list, $arg_list) = $r;
  256.  
  257.         // inifile
  258.         if (isset($opt_list['inifile'])
  259.             && is_readable(end($opt_list['inifile']))) {
  260.             $ini = parse_ini_file(end($opt_list['inifile']), true);
  261.         } else {
  262.             return Ethna::raiseError('give a valid inifile.');
  263.         }
  264.  
  265.         // skelfile
  266.         if (isset($opt_list['skelfile'])
  267.             && is_readable(end($opt_list['skelfile']))) {
  268.             $skelfile = end($opt_list['skelfile']);
  269.         } else {
  270.             return Ethna::raiseError('give a valid filename of plugin skelton file.');
  271.         }
  272.  
  273.         // workdir
  274.         if (isset($opt_list['workdir'])) {
  275.             $workdir = end($opt_list['workdir']);
  276.         } else {
  277.             $workdir = getcwd();
  278.         }
  279.  
  280.         return array($ini, $skelfile, $workdir);
  281.     }
