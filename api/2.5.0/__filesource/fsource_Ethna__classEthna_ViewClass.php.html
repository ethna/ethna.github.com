  356.         case FORM_TYPE_PASSWORD:

  427.         
  428.         // 選択肢を取得

  435.                 $options = $af->$method_or_property();

  444.                 foreach ($attr_list as $key => $val) {
  445.                     $options[$key] = $val['name'];
  446.                 }
  447.             }
  448.         }
  449.  
  450.         if (is_array($options) === false) {
  451.             $this->logger->log(LOG_WARNING,
  452.                 'selector option is not valid. [actionform=%s, option=%s]',
  453.                 get_class($af), $source);
  454.             return null;

  457.         return $options;
  458.     }
