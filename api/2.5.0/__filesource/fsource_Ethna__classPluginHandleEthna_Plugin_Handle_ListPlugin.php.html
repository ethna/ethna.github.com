  123.     // {{{ _getInstalledPluginList()
  124.     /**
  125.      *  get a list of plugins under pear installation management.
  126.      *
  127.      *  @param  object  $pear   Ethna_PearWrapper object.

  133.     function &_getInstalledPluginList(&$pear, $_type = null)
  134.     {
  135.         $pkg_list =& $pear->getInstalledPackageList();
  136.         if (Ethna::isError($pkg_list)) {

  139.  
  140.         $ret = array();
  141.  
  142.         $plugin =& $pear->target_ctl->getPlugin();
  143.         $appid = $pear->target_ctl->getAppId();
  144.         $test_prefix = $pear->target == 'master' ? 'Ethna' : 'App';
  145.  
  146.         foreach ($pkg_list as $pkg_name) {
  147.             list($prefix,, $type, $name) = explode('_', $pkg_name, 4);
  148.             if (($_type === null || $_type == $type) && $prefix == $test_prefix) {
  149.                 list($class_name,,) = $plugin->getPluginNaming($type, $name, $appid);
  150.                 $pkg_version = $pear->getVersion($pkg_name);
  151.                 $pkg_state = $pear->getState($pkg_name);
  152.                 $ret[$class_name] = array($type, $name, $pkg_name,
  153.                                           $pkg_version, $pkg_state);

  161.     /**
  162.      *  get a list of plugins found from controller.
  163.      *  (a local plugin might be installed but still in only skelton.)
  164.      *
  165.      *  @param  object  $pear   Ethna_PearWrapper object.
  166.      *  @param  string  $_type   plugin type

  170.     function &_getFoundPluginList(&$pear, $_type = null)
  171.     {
  172.         $ret = array();
  173.  
  174.         $plugin =& $pear->target_ctl->getPlugin();
  175.         $type_list = $_type === null ? $plugin->searchAllPluginType() : array($_type);
  176.  
  177.         foreach ($type_list as $type) {
  178.             $plugin->searchAllPluginSrc($type);
  179.             if (isset($plugin->src_registry[$type]) === false) {
  180.                 continue;
  181.             }
  182.             foreach ($plugin->src_registry[$type] as $name => $src) {
  183.                 if (empty($src)) {
  184.                     continue;
  185.                 }
  186.                 list($appid,, $type, $name) = explode('_', $src[0], 4);
  187.                 if (($pear->target == 'master' && $appid == 'Ethna') || $appid != 'Ethna') {
  188.                     // XXX: src is private! ([0] is class name)
  189.                     $ret[$src[0]] = array($type, $name);
  190.                 }
  191.             }
  192.         }
  193.         return $ret;
  194.     }
