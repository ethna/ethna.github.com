  31.      *  @access public

  55.

  71.

  98.         // 配列の required_num のチェック

  100.         if (isset($params['num'])) {

  102.                 if (isset($params['error'])) {
  103.                     $msg = $params['error'];
  104.                 } else if ($choice) {

  110.             }

  112.  
  113.         // とくに指定がないとき: フォームに与えられた全要素に

  115.         if (isset($params['key']) == false && isset($params['num']) == false) {

  119.                 } else if ($choice) {
