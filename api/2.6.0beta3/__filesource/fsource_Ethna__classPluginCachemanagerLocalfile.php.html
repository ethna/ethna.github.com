  42.         $cache_file = $this->_getCacheFile($namespace, $key);
  43.  
  44.         // ライフタイムチェック
  45.         clearstatcache();

  47.             || ($st = stat($cache_file)) === false) {

  49.         }

  61.         $timeout = 3;
  62.         while ($timeout > 0) {
  63.             $r = flock($fp, LOCK_EX|LOCK_NB);
  64.             if ($r) {
  65.                 break;
  66.             }
  67.             $timeout--;
  68.             sleep(1);
  69.         }
  70.         if ($timeout <= 0) {
  71.             fclose($fp);

  154.     public function set($key, $value, $timestamp = null, $namespace = null)
  155.     {

  158.  
  159.         // キャッシュディレクトリチェック
  160.         $r = Ethna_Util::mkdir($dir, 0777);
  161.         if ($r == false && is_dir($dir) == false) {
  162.             return Ethna::raiseError('mkdir(%s) failed', E_USER_WARNING, $dir);
  163.         }
  164.  
  165.         $cache_file = $this->_getCacheFile($namespace, $key);
  166.         $fp = fopen($cache_file, "a+");

  168.             return Ethna::raiseError('fopen failed', E_CACHE_GENERAL);
  169.         }
  170.  
  171.         // ロック
  172.         $timeout = 3;
  173.         while ($timeout > 0) {
  174.             $r = flock($fp, LOCK_EX|LOCK_NB);
  175.             if ($r) {

  181.         if ($timeout <= 0) {

  191.         if (is_null($timestamp)) {
  192.             // this could suppress warning
  193.             touch($cache_file);
  194.         } else {
  195.             touch($cache_file, $timestamp);
  196.         }
  197.  
  198.         return 0;
  199.     }

  223.     private function _getCacheDir($namespace, $key)
  224.     {
  225.         $safe_mode = ini_get('safe_mode');

  241.         //$map = $this->config->get('cachemanager_localfile');

  243.         $tmp_key = $namespace . "::" . $key;

  246.  
  247.         if (is_array($map)) {

  253.             }
  254.         }
  255.         return sprintf("%s/cache/%s/cache_%s/%s/%s", $this->backend->getTmpdir(), $dir, $this->_escape($namespace), $this->_escape($dir1), $this->_escape($dir2));
  256.     }
