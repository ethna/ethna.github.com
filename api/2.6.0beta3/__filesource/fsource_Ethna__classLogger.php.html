  281.     /**

  288.     public function log($level, $message)
  289.     {

  299.  
  300.         foreach (array_keys($this->writer) as $key) {
  301.             // ログメッセージフィルタ(レベルフィルタに優先する)
  302.             $r = $this->_evalMessageMask($this->message_filter_do[$key], $message);
  303.             if (is_null($r)) {
  304.                 $r = $this->_evalMessageMask($this->message_filter_ignore[$key],
  305.                                              $message);
  306.                 if ($r) {
  307.                     continue;
  308.                 }
  309.             }
  310.  
  311.             // ログレベルフィルタ
  312.             if ($this->_evalLevelMask($this->level[$key], $level)) {
  313.                 continue;
  314.             }
  315.  
  316.             // ログ出力
  317.             $args = func_get_args();
  318.             if (count($args) > 2) {
  319.                 array_splice($args, 0, 2);
  320.                 $message = vsprintf($message, $args);
  321.             }

  323.         }
  324.

  327.             if (count($this->alert_mailaddress) > 0) {
  328.                 $this->_alert($output);
  329.             }
  330.         }

  384.         if (Ethna::isError($plugin_object)) {

  387.  
  388.         if (isset($option['ident']) == false) {
  389.             $option['ident'] = $this->controller->getAppId();
  390.         }
  391.         if (isset($option['facility']) == false) {
  392.             $option['facility'] = $facility;
  393.         }
  394.         $plugin_object->setOption($option);
  395.  
  396.         return $plugin_object;
  397.     }

  421.         
  422.         // 本文

  424.         if (function_exists("debug_backtrace")) {
