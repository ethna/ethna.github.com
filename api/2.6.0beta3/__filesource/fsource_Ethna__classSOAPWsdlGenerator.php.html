  121.     function _getTypes()

  123.         $types = sprintf(" <types>\n  <s:schema targetNamespace=\"%s\">\n", $this->namespace);
  124.  
  125.         // 基本型
  126.         $types .=<<<EOD
  127.    <s:complexType name="ArrayOfInt">
  128.     <s:complexContent mixed="false">
  129.      <s:restriction base="soapenc:Array">
  130.       <s:attribute d7p1:arrayType="s:int[]" ref="soapenc:arrayType" xmlns:d7p1="http://schemas.xmlsoap.org/wsdl/" />
  131.      </s:restriction>
  132.     </s:complexContent>
  133.    </s:complexType>
  134.    <s:complexType name="ArrayOfString">
  135.     <s:complexContent mixed="false">
  136.      <s:restriction base="soapenc:Array">
  137.       <s:attribute d7p1:arrayType="s:string[]" ref="soapenc:arrayType" xmlns:d7p1="http://schemas.xmlsoap.org/wsdl/" />
  138.      </s:restriction>
  139.     </s:complexContent>
  140.    </s:complexType>
  141.    <s:complexType name="Result">
  142.     <s:sequence>
  143.      <s:element name="errormessage" type="s:string" />
  144.      <s:element name="errorcode" type="s:int" />

  148.         
  149.         // アクション固有
  150.         foreach ($this->controller->soap_action as $k => $v) {

  152.             $form = new $action_form_name($this->controller);
  153.             if ($form->retval == null) {
  154.                 continue;
  155.             }
  156.  
  157.             // デフォルトエントリを追加
  158.             Ethna_SOAP_Util::fixRetval($form->retval);
  159.  
  160.             // シリアライズ
  161.             $retval_name = preg_replace('/_(.)/e', "strtoupper('\$1')", ucfirst($k)) . "Result";
  162.             $types .= $this->_serializeTypes($form->retval, $retval_name);
  163.         }
  164.  
  165.         return $types . "  </s:schema>\n </types>\n\n";
  166.     }

  279.     /**
  280.      *  型のシリアライズ
  281.      *
  282.      *  @access private
  283.      *  @param  array   $def    型定義
  284.      *  @param  string  $name   変数名
  285.      *  @return string  シリアライズされた型定義
  286.      */
  287.     function _serializeTypes($def, $name)
  288.     {
  289.         if (is_array($def) == false) {
  290.             // nothing to do
  291.             return;
  292.         }
  293.  
  294.         $types = $this->__serializeTypes($def, $name);
  295.  
  296.         foreach ($def as $k => $v) {
  297.             if (is_array($def[$k]) == false || Ethna_SOAP_Util::isArrayOfScalar($def[$k])) {
  298.                 continue;
  299.             }
  300.             $types .= $this->_serializeTypes($def[$k], $k);
  301.         }
  302.  
  303.         return $types;
  304.     }

  306.     /**
  307.      *  型のシリアライズ(エレメント対応)
  308.      *
  309.      *  @access private
  310.      *  @param  array   $def    型定義
  311.      *  @param  string  $name   変数名
  312.      *  @return string  シリアライズされた型定義(各要素)
  313.      */
  314.     function __serializeTypes($def, $name)
  315.     {
  316.         $keys = array_keys($def);
  317.  
  318.         if (Ethna_SOAP_Util::isArrayOfObject($def)) {
  319.             $array_name = sprintf("ArrayOf%s", $keys[0]);
  320.             $name = $keys[0];
  321.             $types = "   <s:complexType name=\"$array_name\">\n";
  322.             $types .= "    <s:complexContent mixed=\"false\">\n";
  323.             $types .= "     <s:restriction base=\"soapenc:Array\">\n";
  324.             $types .= "      <s:attribute d7p1:arrayType=\"tns:$name" . "[]" . "\" " .
  325.                 "ref=\"soapenc:arrayType\" xmlns:d7p1=\"http://schemas.xmlsoap.org/wsdl/\" />\n";
  326.             $types .= "     </s:restriction>\n";
  327.             $types .= "    </s:complexContent>\n";
  328.             $types .= "   </s:complexType>\n";
  329.             return $types;
  330.         }
  331.  
  332.         $types = "   <s:complexType name=\"$name\">\n";
  333.         $types .= "    <s:sequence>\n";
  334.         foreach ($keys as $key) {
  335.             if (is_array($def[$key])) {
  336.                 $inner_keys = array_keys($def[$key]);
  337.                 if (is_array($def[$key][$inner_keys[0]])) {
  338.                     $inner_name = sprintf("ArrayOf%s", $inner_keys[0]);
  339.                     $types .= "     <s:element name=\"$key\" type=\"tns:$inner_name\" />\n";
  340.                 } else {
  341.                     $type_name = "tns:" . Ethna_SOAP_Util::getArrayTypeName($def[$key][$inner_keys[0]]);
  342.                     $types .= "     <s:element name=\"$key\" type=\"$type_name\" />\n";
  343.                 }
  344.             } else {
  345.                 $type_name = Ethna_SOAP_Util::getScalarTypeName($def[$key]);
  346.                 $types .= "     <s:element name=\"$key\" type=\"s:$type_name\" />\n";
  347.             }
  348.         }
  349.         $types .= "    </s:sequence>\n";
  350.         $types .= "   </s:complexType>\n";
  351.  
  352.         return $types;
  353.     }

  355.     /**
  356.      *  Messageのシリアライズ
  357.      *
  358.      *  @access private
  359.      *  @param  string  $name   message名
  360.      *  @param  int     $serno  message連番
  361.      *  @return string  シリアライズされたmessage
  362.      */
  363.     function _serializeMessage($name, $serno)
  364.     {
  365.         $action_form_name = $this->controller->getActionFormName($name);
  366.         $form = new $action_form_name($this->controller);
  367.  
  368.         /* SoapIn */
  369.         $message = " <message name=\"${name}${serno}SoapIn\">\n";
  370.         $keys = array();
  371.         if (is_array($form->form)) {
  372.             $keys = array_keys($form->form);
  373.         }
  374.         foreach ($keys as $key) {
  375.             $type_id = $form->form[$key]['type'];
  376.             if (is_array($type_id)) {
  377.                 $type_keys = array_keys($type_id);
  378.                 $type = "tns:" . Ethna_SOAP_Util::getArrayTypeName($type_id[$type_keys[0]]);
  379.             } else {
  380.                 $type = "s:" . Ethna_SOAP_Util::getScalarTypeName($type_id);
  381.             }
  382.             $message .= "  <part name=\"$key\" type=\"$type\" />\n";
  383.         }
  384.         $message .= " </message>\n";
  385.  
  386.         /* SoapOut */
  387.         $message .= " <message name=\"${name}${serno}SoapOut\">\n";
  388.         if ($form->retval == null) {
  389.             $type = "tns:Result";
  390.         } else {
  391.             $type = "tns:${name}Result";
  392.         }
  393.         $message .= "  <part name=\"result\" type=\"$type\" />\n";
  394.         $message .= " </message>\n";
  395.  
  396.         return $message;
  397.     }

  399.     /**
  400.      *  PortTypeのシリアライズ
  401.      *
  402.      *  @access private
  403.      *  @param  string  $name   porttype名
  404.      *  @param  int     $serno  porttype連番
  405.      *  @return string  シリアライズされたporttype
  406.      */
  407.     function _serializePortType($name, $serno)
  408.     {
  409.         $action_form_name = $this->controller->getActionFormName($name);
  410.         $form = new $action_form_name($this->controller);
  411.  
  412.         $args = null;
  413.         if (is_array($form->form)) {
  414.             $args = implode(' ', array_keys($form->form));
  415.         }
  416.  
  417.         $port_type = "  <operation name=\"$name\" parameterOrder=\"$args\">\n";
  418.         $port_type .= "   <input name=\"${name}${serno}SoapIn\" message=\"tns:${name}${serno}SoapIn\" />\n";
  419.         $port_type .= "   <output name=\"${name}${serno}SoapOut\" message=\"tns:${name}${serno}SoapOut\" />\n";
  420.         $port_type .= "  </operation>\n";
  421.  
  422.         return $port_type;
  423.     }
