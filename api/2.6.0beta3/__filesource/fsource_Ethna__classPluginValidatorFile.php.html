  177.     function _getSizeAsBytes($size)

  181.             if (isset($matches[1])) {
  182.                 $size = $matches[1];
  183.             }
  184.             if (isset($matches[2])) {
  185.                 if (strtolower($matches[2]) === 'm') {
  186.                     $unit = 1048576;
