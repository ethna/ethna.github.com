  70.             if (count($inifile) > 1) {
  71.                 return Ethna::raiseError(
  72.                     'more than 2 .ini file found.'
  73.                   . ' please specify [-i|--ini-file-path] option',
  74.                     'usage'

  96.     function perform()
  97.     {

  102.                 "ERROR: PEAR_PackageFileManager2 is not installed! please install it.\n"

  106.  
  107.         require_once 'PEAR/PackageFileManager2.php';

  109.  
  110.         // 引数の評価
  111.         $args = $this->_parseArgList();
  112.         if (Ethna::isError($args)) {
  113.             return $args;
  114.         }
  115.         list($ini, $plugin_dir) = $args;
  116.  
  117.         // 設定用の配列を用意
  118.         $setting = array();
  119.  
  120.         // プラグイン名
  121.         $setting['pkgname'] = $ini['plugin']['name'];
  122.  
  123.         // パッケージの説明

  125.         $setting['summary']     = $ini['package']['summary'];
  126.         $setting['description'] = $ini['package']['description'];
  127.  
  128.         // リリースの説明

  132.  
  133.         // メンテナ

  135.         for ($i = 1; ; $i++) {
  136.             if (isset($mnts["name$i"]) == false
  137.                 || isset($mnts["user$i"]) == false
  138.                 || isset($mnts["email$i"]) == false) {
  139.                 break;

  142.                 'user'   => $mnts["user$i"],
  143.                 'name'   => $mnts["name$i"],
  144.                 'email'  => $mnts["email$i"],
  145.                 'role'   => isset($mnts["role$i"]) ? $mnts["role$i"] : 'lead',
  146.                 'active' => isset($mnts["active$i"]) ? $mnts["active$i"] == true: 'yes',
  147.                 );
  148.         }
  149.  
  150.         // ライセンス
  151.         $setting['license']['name'] = $ini['license']['name'];
  152.         if (isset($ini['license']['uri'])) {
  153.             $setting['license']['uri'] = $ini['license']['uri'];
  154.         }

  173.     /**
  174.      * @access private
  175.      */
  176.     function _makePackage($setting, $workdir)
  177.     {
  178.         // package.xml を作る
  179.         $pkgconfig = array(
  180.             'packagedirectory' => $workdir,
  181.             'outputdirectory' => $workdir,
  182.             'ignore' => array('CVS/', '.cvsignore', '.svn/',
  183.                               'package.xml', '*.ini', $setting['pkgname'].'-*.tgz'),
  184.             'filelistgenerator' => 'file',
  185.             'changelogoldtonew' => false,
  186.             );
  187.  
  188.         $packagexml = new PEAR_PackageFileManager2();
  189.  
  190.         $pkgconfig = array_merge($pkgconfig, $setting['config']);
  191.         $packagexml->setOptions($pkgconfig);
  192.  
  193.         $packagexml->setPackage($setting['pkgname']);
  194.         $packagexml->setSummary($setting['summary']);
  195.         $packagexml->setNotes($setting['notes']);
  196.         $packagexml->setDescription($setting['description']);
  197.         $packagexml->setChannel($setting['channel']);
  198.         $packagexml->setAPIVersion($setting['version']);
  199.         $packagexml->setReleaseVersion($setting['version']);
  200.         $packagexml->setReleaseStability($setting['state']);
  201.         $packagexml->setAPIStability($setting['state']);
  202.         $packagexml->setPackageType('php');
  203.         foreach ($setting['maintainers'] as $m) {
  204.             $packagexml->addMaintainer($m['role'], $m['user'], $m['name'],
  205.                                        $m['email'], $m['active']);
  206.         }
  207.         $packagexml->setLicense($setting['license']['name'],
  208.                                 $setting['license']['uri']);
  209.  
  210.         $packagexml->addRole('css', 'php');
  211.         $packagexml->addRole('tpl', 'php');
  212.         $packagexml->addRole('ethna', 'php');
  213.         $packagexml->addRole('sh', 'script');
  214.         $packagexml->addRole('bat', 'script');
  215.  
  216.         $packagexml->setPhpDep('4.3.0');
  217.         $packagexml->setPearinstallerDep('1.3.5');
  218.  
  219.         $packagexml->generateContents();
  220.  
  221.         foreach ($setting['callback'] as $method => $params) {
  222.             $r = call_user_func_array(array($packagexml, $method), $params);
  223.         }
  224.  
  225.         $r = $packagexml->writePackageFile();
  226.         if (PEAR::isError($r)) {
  227.             return Ethna::raiseError($r->getMessage, $r->getCode());
  228.         }
  229.  
  230.         //  finally make package
  231.         PEAR_Command::setFrontendType('CLI');
  232.         $ui = PEAR_Command::getFrontendObject();
  233.         $config = PEAR_Config::singleton();
  234.         $ui->setConfig($config);
  235.         PEAR::setErrorHandling(PEAR_ERROR_CALLBACK, array($ui, 'displayFatalError'));
  236.         $cmd = PEAR_Command::factory('package', $config);
  237.         if (PEAR::isError($cmd)) {
  238.             return Ethna::raiseError($cmd->getMessage, $cmd->getCode());
  239.         }
  240.         $r = $cmd->run('package', array(), array("$workdir/package.xml"));
  241.         if (PEAR::isError($r)) {
  242.             return Ethna::raiseError($r->getMessage, $r->getCode());
  243.         }
  244.     }
