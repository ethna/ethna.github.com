  161.     {
  162.         $r = array();
  163.         $cache_update = false;
  164.  
  165.         if (is_null($action_dir)) {
  166.             $cache_update = true;
  167.             $action_dir = $this->ctl->getActiondir();

  170.  
  171.         $child_dir_list = array();
  172.  
  173.         $dh = opendir($action_dir);
  174.         if ($dh == false) {

  176.         }
  177.  
  178.         $ext = $this->ctl->getExt('php');
  179.         while (($file = readdir($dh)) !== false) {
  180.             if ($file == "." || $file == "..") {
  181.                 continue;
  182.             }
  183.             $file = $action_dir . $file;
  184.  
  185.             if (is_dir($file)) {
  186.                 $child_dir_list[] = $file;

  188.             }
  189.  
  190.             if (preg_match("/\.$ext\$/", $file) == 0) {
  191.                 continue;

  193.  
  194.             $key = substr($file, $prefix_len);
  195.             
  196.             // キャッシュチェック
  197.             include_once $file;
  198.             if ($this->cache_class_list[$key]['.mtime'] >= filemtime($file)) {
  199.                 $class_list = $this->cache_class_list[$key];
  200.             } else {
  201.                 $class_list = $this->_analyzeActionScript($file);
  202.             }
  203.             if (is_null($class_list) == false) {
  204.                 $r[$key] = $class_list;
  205.             }
  206.         }
  207.  
  208.         closedir($dh);
  209.  
  210.         foreach ($child_dir_list as $child_dir) {
  211.             $tmp = $this->_analyzeActionList($child_dir . "/");
  212.             $r = array_merge($r, $tmp);
  213.         }
  214.  
  215.         if ($cache_update) {
  216.             // キャッシュファイル更新
  217.             $fp = fopen($this->cache_class_list_file, 'w');
  218.             fwrite($fp, serialize($r));
  219.             fclose($fp);
  220.         }
  221.  
  222.         return $r;
  223.     }

  225.     /**
  226.      *  アクションスクリプトを解析する
  227.      *
  228.      *  @access private
  229.      *  @param  string  $script ファイル名
  230.      *  @return array   アクションクラス定義一覧
  231.      */
  232.     function _analyzeActionScript($script)
  233.     {
  234.         $class_list = array();
  235.         $class_list['.mtime'] = filemtime($script);
  236.  
  237.         $source = "";
  238.         $fp = fopen($script, 'r');
  239.         if ($fp == false) {
  240.             return null;
  241.         }
  242.         while (feof($fp) == false) {
  243.             $source .= fgets($fp, 8192);
  244.         }
  245.         fclose($fp);
  246.  
  247.         // トークンに分割してクラス定義情報を取得
  248.         $token_list = token_get_all($source);
  249.         $state = 'T_OUT';
  250.         $nest = 0;
  251.         $method_nest = 0;
  252.         $current = null;
  253.         for ($i = 0; $i < count($token_list); $i++) {
  254.             $token = $token_list[$i];
  255.  
  256.             if (is_string($token)) {
  257.                 if ($token == '{') {
  258.                     $nest++;
  259.                 } else if ($token == '}') {
  260.                     $nest--;
  261.                     if ($state == 'T_PREPARE' || $state == 'T_PERFORM') {
  262.                         if ($nest == $method_nest) {
  263.                             $state = 'T_ACTION_CLASS';
  264.                         }
  265.                     } else if ($nest == 0) {
  266.                         $state = 'T_OUT';
  267.                     }
  268.                 }
  269.                 continue;
  270.             }
  271.  
  272.             if ($token[0] == T_CLASS) {
  273.                 // クラス定義開始
  274.                 $i += 2;
  275.                 $class_name = $token_list[$i][1];       // should be T_STRING
  276.                 if ($this->_isSubclassOf($class_name, 'Ethna_ActionClass')) {
  277.                     $state = 'T_ACTION_CLASS';
  278.                     $current = $class_name;
  279.                     $class_list[$current] = array('type' => 'action_class');
  280.                 } else if ($this->_isSubclassOf($class_name, 'Ethna_ActionForm')) {
  281.                     $state = 'T_ACTION_FORM';
  282.                     $current = $class_name;
  283.                     $class_list[$current] = array('type' => 'action_form');
  284.                 }
  285.                 $nest = 0;  // for safe
  286.             } else if ($token[0] == T_COMMENT && strncmp($token[1], "/**", 3) == 0 && is_array($token_list[$i+2]) && $token_list[$i+2][0] == T_CLASS) {
  287.                 // DocComment for class
  288.             } else if ($state == 'T_ACTION_CLASS' && $token[0] == T_FUNCTION) {
  289.                 $i += 2;
  290.                 $method_name = $token_list[$i][1];
  291.                 if (strcasecmp($method_name, 'prepare') == 0) {
  292.                     $state = 'T_PREPARE';
  293.                     $method_nest = $nest;
  294.                 } else if (strcasecmp($method_name, 'perform') == 0) {
  295.                     $state = 'T_PERFORM';
  296.                     $method_nest = $nest;
  297.                 }
  298.             } else if (($state == 'T_PREPARE' || $state == 'T_PERFORM') && $token[0] == T_RETURN) {
  299.                 $s = "";
  300.                 $n = 1;
  301.                 while ($token_list[$i+$n] !== ";") {
  302.                     $s .= is_string($token_list[$i+$n]) ? $token_list[$i+$n] : $token_list[$i+$n][1];
  303.                     $n++;
  304.                 }
  305.                 $key = $state == 'T_PREPARE' ? 'prepare' : 'perform';
  306.                 $class_list[$current]['return'][$key][] = $s;
  307.             }
  308.         }
  309.  
  310.         if (count($class_list) == 0) {
  311.             return null;
  312.         }
  313.         return $class_list;
  314.     }

  316.     /**
  317.      *  指定されたクラス名を継承しているかどうかを返す
  318.      *
  319.      *  @access private
  320.      *  @param  string  $class_name     チェック対象のクラス名
  321.      *  @param  string  $parent_name    親クラス名
  322.      *  @return bool    true:継承している false:いない

  324.     function _isSubclassOf($class_name, $parent_name)
  325.     {
  326.         while ($tmp = get_parent_class($class_name)) {
  327.             if (strcasecmp($tmp, $parent_name) == 0) {
  328.                 return true;
  329.             }
  330.             $class_name = $tmp;
  331.         }
  332.         return false;
  333.     }

  336.      *  コントローラに明示的に定義されているアクション一覧を取得する

  342.     function _getActionList_Manifest($class_list)
  343.     {
  344.         $manifest_action_list = array();
  345.         $manifest_class_list = array();
  346.         foreach ($this->_getInvisibleProperty($this->ctl, 'action') as $action_name => $action) {
  347.             if ($action_name == '__ethna_info__' || $action_name == '__ethna_unittest__') {
  348.                 continue;
  349.             }
  350.             $action = $this->ctl->_getAction($action_name);
  351.  
  352.             $elt = array();
  353.             // _analyzeActionList()で取得したクラス定義データから対応関係を取得
  354.             foreach ($class_list as $file => $elts) {
  355.                 foreach ($elts as $class_name => $def) {
  356.                     if ($def['type'] == 'action_class' && strcasecmp($class_name, $action['class_name']) == 0) {
  357.                         $elt['action_class'] = $class_name;
  358.                         $elt['action_class_file'] = $file;
  359.                         $elt['action_class_info'] = $def;
  360.                     } else if ($def['type'] == 'action_form' && strcasecmp($class_name, $action['form_name']) == 0) {
  361.                         $elt['action_form'] = $class_name;
  362.                         $elt['action_form_file'] = $file;
  363.                         $elt['action_form_info'] = $def;
  364.                     }
  365.                 }
  366.             }
  367.  
  368.             // 未定義チェック
  369.             if (isset($elt['action_class']) == false) {
  370.                 $elt['action_class'] = $action['class_name'];
  371.                 if (class_exists($action['class_name']) == false) {
  372.                     $elt['action_class_info'] = array('undef' => true);
  373.                 }
  374.                 else {
  375.                     $elt['action_class_info'] = array('undef' => false);
  376.                 }
  377.             }
  378.  
  379.             if (isset($elt['action_form']) == false && $action['form_name'] != 'Ethna_ActionForm') {
  380.                 $elt['action_form'] = $action['form_name'];
  381.                 if (class_exists($action['form_name']) == false) {
  382.                     $elt['action_form_info'] = array('undef' => true);
  383.                 }
  384.                 else {
  385.                     $elt['action_form_info'] = array('undef' => false);
  386.                 }
  387.             }
  388.             $manifest_action_list[$action_name] = $elt;
  389.             $manifest_class_list[] = strtolower($elt['action_class']);
  390.         }
  391.  
  392.         return array($manifest_action_list, $manifest_class_list);
  393.     }

  395.     /**
  396.      *  暗黙に定義されているアクション一覧を取得する
  397.      *
  398.      *  @access private
  399.      *  @param  array   $class_list             定義されているクラス一覧
  400.      *  @param  array   $manifest_action_list   明示的に定義済みのアクション一覧
  401.      *  @param  array   $manifest_class_list    明示的に定義済みのクラス一覧
  402.      *  @return array   array:アクション一覧
  403.      */
  404.     function _getActionList_Implicit($class_list, $manifest_action_list, $manifest_class_list)
  405.     {
  406.         $implicit_action_list = array();
  407.  
  408.         foreach ($class_list as $file => $elts) {
  409.             foreach ($elts as $class_name => $def) {
  410.                 if (in_array(strtolower($class_name), $manifest_class_list)) {
  411.                     continue;
  412.                 }
  413.  
  414.                 // クラス名からアクション名を取得
  415.                 if ($def['type'] == 'action_class') {
  416.                     $action_name = $this->ctl->actionClassToName($class_name);
  417.                     if (array_key_exists($action_name, $manifest_action_list)) {

  484.     {
  485.         $r = array();
  486.  
  487.         if (is_null($template_dir)) {
  488.             $template_dir = $this->ctl->getTemplatedir();
  489.         }

  491.  
  492.         $child_dir_list = array();
  493.  
  494.         $dh = opendir($template_dir);
  495.         if ($dh == false) {
  496.             return;
  497.         }
  498.  
  499.         $ext = $this->ctl->getExt('tpl');

  502.                 continue;
  503.             }
  504.             $file = $template_dir . '/' . $file;
  505.  
  506.             if (is_dir($file)) {

  512.                 continue;

  518.  
  519.         closedir($dh);
  520.  
  521.         foreach ($child_dir_list as $child_dir) {
  522.             $tmp = $this->_analyzeForwardList($child_dir);
